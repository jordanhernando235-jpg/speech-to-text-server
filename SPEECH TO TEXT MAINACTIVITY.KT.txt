package com.example.speechtotextapp

import android.content.ClipData
import android.content.ClipboardManager
import android.content.Context
import android.content.Intent
import android.net.Uri
import android.os.Bundle
import android.speech.tts.TextToSpeech
import android.widget.Toast
import androidx.activity.ComponentActivity
import androidx.activity.compose.setContent
import androidx.activity.result.contract.ActivityResultContracts
import androidx.compose.foundation.layout.*
import androidx.compose.foundation.rememberScrollState
import androidx.compose.foundation.verticalScroll
import androidx.compose.material3.*
import androidx.compose.runtime.*
import androidx.compose.ui.Modifier
import androidx.compose.ui.text.TextStyle
import androidx.compose.ui.unit.dp
import androidx.compose.ui.unit.sp
import okhttp3.*
import okhttp3.MediaType.Companion.toMediaTypeOrNull
import okhttp3.RequestBody
import org.json.JSONObject
import java.io.IOException
import java.net.URLEncoder
import java.util.Locale
import java.util.concurrent.TimeUnit

class MainActivity :
    ComponentActivity(),
    TextToSpeech.OnInitListener {

    private lateinit var textToSpeech: TextToSpeech

    private val resultText =
        mutableStateOf("Pick audio file to start")

    private val uploadProgress =
        mutableStateOf(0)

    // YOUR RAILWAY URL
    private val BASE_URL =
        "https://fool-vaporizer-alibi.ngrok-free.dev"

    private val picker =
        registerForActivityResult(
            ActivityResultContracts.GetContent()
        ) { uri: Uri? ->

            uri?.let {
                uploadAudio(it)
            }
        }

    override fun onCreate(savedInstanceState: Bundle?) {

        super.onCreate(savedInstanceState)

        textToSpeech = TextToSpeech(this, this)

        setContent {

            val scrollState =
                rememberScrollState()

            MaterialTheme {

                Column(
                    modifier = Modifier
                        .fillMaxSize()
                        .padding(16.dp)
                ) {

                    // PICK AUDIO
                    Button(
                        onClick = {
                            picker.launch("audio/*")
                        },
                        modifier = Modifier.fillMaxWidth()
                    ) {
                        Text("Pick Audio")
                    }

                    Spacer(
                        modifier = Modifier.height(8.dp)
                    )

                    // SERVER URL
                    Text(
                        text = "Server URL:\n$BASE_URL",
                        fontSize = 12.sp
                    )

                    Spacer(
                        modifier = Modifier.height(8.dp)
                    )

                    // URL BUTTONS
                    Row(
                        modifier = Modifier.fillMaxWidth(),
                        horizontalArrangement = Arrangement.spacedBy(8.dp)
                    ) {

                        // COPY URL
                        Button(
                            onClick = {

                                val clipboard =
                                    getSystemService(
                                        Context.CLIPBOARD_SERVICE
                                    ) as ClipboardManager

                                val clip =
                                    ClipData.newPlainText(
                                        "Server URL",
                                        BASE_URL
                                    )

                                clipboard.setPrimaryClip(clip)

                                Toast.makeText(
                                    this@MainActivity,
                                    "URL copied!",
                                    Toast.LENGTH_SHORT
                                ).show()
                            },
                            modifier = Modifier.weight(1f)
                        ) {
                            Text("Copy URL")
                        }

                        // OPEN URL
                        Button(
                            onClick = {

                                val intent =
                                    Intent(
                                        Intent.ACTION_VIEW,
                                        Uri.parse(BASE_URL)
                                    )

                                startActivity(intent)
                            },
                            modifier = Modifier.weight(1f)
                        ) {
                            Text("Open URL")
                        }
                    }

                    Spacer(
                        modifier = Modifier.height(16.dp)
                    )

                    // PROGRESS BAR
                    LinearProgressIndicator(
                        progress = {
                            uploadProgress.value / 100f
                        },
                        modifier = Modifier.fillMaxWidth()
                    )

                    Spacer(
                        modifier = Modifier.height(8.dp)
                    )

                    Text(
                        text = "Upload: ${uploadProgress.value}%",
                        fontSize = 14.sp
                    )

                    Spacer(
                        modifier = Modifier.height(16.dp)
                    )

                    // LANGUAGE SELECT
                    var targetLanguage by remember {
                        mutableStateOf("id")
                    }

                    val languages = mapOf(
                        "English" to "en",
                        "Indonesian" to "id",
                        "Spanish" to "es",
                        "French" to "fr",
                        "German" to "de",
                        "Japanese" to "ja",
                        "Korean" to "ko",
                        "Chinese" to "zh"
                    )

                    var targetExpanded by remember {
                        mutableStateOf(false)
                    }

                    Box {

                        Button(
                            onClick = {
                                targetExpanded = true
                            },
                            modifier = Modifier.fillMaxWidth()
                        ) {

                            Text(
                                "Translate To: ${
                                    languages.entries.find {
                                        it.value == targetLanguage
                                    }?.key
                                }"
                            )
                        }

                        DropdownMenu(
                            expanded = targetExpanded,
                            onDismissRequest = {
                                targetExpanded = false
                            }
                        ) {

                            languages.forEach { (name, code) ->

                                DropdownMenuItem(
                                    text = {
                                        Text(name)
                                    },
                                    onClick = {

                                        targetLanguage = code

                                        targetExpanded = false
                                    }
                                )
                            }
                        }
                    }

                    Spacer(
                        modifier = Modifier.height(12.dp)
                    )

                    // TRANSLATE BUTTON
                    Button(
                        onClick = {
                            translateText(targetLanguage)
                        },
                        modifier = Modifier.fillMaxWidth()
                    ) {
                        Text("Translate")
                    }

                    Spacer(
                        modifier = Modifier.height(12.dp)
                    )

                    // COPY + SPEAK BUTTONS
                    Row(
                        modifier = Modifier.fillMaxWidth(),
                        horizontalArrangement = Arrangement.spacedBy(8.dp)
                    ) {

                        // COPY TEXT
                        Button(
                            onClick = {

                                val clipboard =
                                    getSystemService(
                                        Context.CLIPBOARD_SERVICE
                                    ) as ClipboardManager

                                val clip =
                                    ClipData.newPlainText(
                                        "Transcription",
                                        resultText.value
                                    )

                                clipboard.setPrimaryClip(clip)

                                Toast.makeText(
                                    this@MainActivity,
                                    "Text copied!",
                                    Toast.LENGTH_SHORT
                                ).show()
                            },
                            modifier = Modifier.weight(1f)
                        ) {
                            Text("Copy Text")
                        }

                        // SPEAK
                        Button(
                            onClick = {

                                textToSpeech.language = Locale.US

                                textToSpeech.speak(
                                    resultText.value,
                                    TextToSpeech.QUEUE_FLUSH,
                                    null,
                                    null
                                )
                            },
                            modifier = Modifier.weight(1f)
                        ) {
                            Text("Speak")
                        }
                    }

                    Spacer(
                        modifier = Modifier.height(12.dp)
                    )

                    // RESULT BOX
                    Card(
                        modifier = Modifier
                            .fillMaxWidth()
                            .height(400.dp)
                    ) {

                        Column(
                            modifier = Modifier
                                .fillMaxSize()
                                .padding(12.dp)
                                .verticalScroll(scrollState)
                        ) {

                            Text(
                                text = "Result",
                                fontSize = 14.sp
                            )

                            Spacer(
                                modifier = Modifier.height(8.dp)
                            )

                            Text(
                                text = resultText.value,
                                style = TextStyle(
                                    fontSize = 16.sp,
                                    lineHeight = 24.sp
                                )
                            )
                        }
                    }
                }
            }
        }
    }

    // UPLOAD AUDIO
    private fun uploadAudio(uri: Uri) {

        resultText.value = "Uploading..."
        uploadProgress.value = 10

        Thread {

            try {

                val inputStream =
                    contentResolver.openInputStream(uri)

                if (inputStream == null) {

                    runOnUiThread {

                        resultText.value =
                            "Cannot open file"
                    }

                    return@Thread
                }

                val bytes =
                    inputStream.readBytes()

                inputStream.close()

                val client =
                    OkHttpClient.Builder()

                        .connectTimeout(
                            60,
                            TimeUnit.SECONDS
                        )

                        .writeTimeout(
                            120,
                            TimeUnit.SECONDS
                        )

                        .readTimeout(
                            120,
                            TimeUnit.SECONDS
                        )

                        .build()

                val requestBody =
                    RequestBody.create(
                        "audio/*"
                            .toMediaTypeOrNull(),
                        bytes
                    )

                val fileName =
                    uri.lastPathSegment ?: "audio"

                val multipartBody =
                    MultipartBody.Builder()

                        .setType(
                            MultipartBody.FORM
                        )

                        .addFormDataPart(
                            "file",
                            fileName,
                            requestBody
                        )

                        .build()

                val request =
                    Request.Builder()

                        .url(
                            "$BASE_URL/transcribe"
                        )

                        .post(multipartBody)

                        .build()

                client.newCall(request)

                    .enqueue(object : Callback {

                        override fun onFailure(
                            call: Call,
                            e: IOException
                        ) {

                            runOnUiThread {

                                resultText.value =
                                    "Upload failed: ${e.message}"

                                uploadProgress.value = 0
                            }
                        }

                        override fun onResponse(
                            call: Call,
                            response: Response
                        ) {

                            val body =
                                response.body?.string()

                            runOnUiThread {

                                uploadProgress.value = 100

                                if (
                                    response.isSuccessful &&
                                    body != null
                                ) {

                                    try {

                                        val json =
                                            JSONObject(body)

                                        val originalText =
                                            json.optString(
                                                "text",
                                                "No text found"
                                            )

                                        val translatedText =
                                            json.optString(
                                                "translated",
                                                originalText
                                            )

                                        resultText.value =
                                            "Original:\n\n$originalText\n\nTranslated:\n\n$translatedText"

                                    } catch (
                                        e: Exception
                                    ) {

                                        resultText.value =
                                            "Parse error: $body"
                                    }

                                } else {

                                    resultText.value =
                                        "Server error: ${response.code}"
                                }
                            }
                        }
                    })

            } catch (e: Exception) {

                runOnUiThread {

                    resultText.value =
                        "Exception: ${e.message}"

                    uploadProgress.value = 0
                }
            }

        }.start()
    }

    // TRANSLATE TEXT
    private fun translateText(to: String) {

        val originalText =
            resultText.value

        runOnUiThread {

            resultText.value =
                "Translating..."
        }

        Thread {

            try {

                val client = OkHttpClient()

                val encodedText =
                    URLEncoder.encode(
                        originalText,
                        "UTF-8"
                    )

                val url =
                    "https://translate.googleapis.com/translate_a/single?client=gtx&sl=auto&tl=$to&dt=t&q=$encodedText"

                val request =
                    Request.Builder()
                        .url(url)
                        .build()

                client.newCall(request)

                    .enqueue(object : Callback {

                        override fun onFailure(
                            call: Call,
                            e: IOException
                        ) {

                            runOnUiThread {

                                resultText.value =
                                    "Translation failed"
                            }
                        }

                        override fun onResponse(
                            call: Call,
                            response: Response
                        ) {

                            val body =
                                response.body?.string()

                            if (body != null) {

                                try {

                                    val translated =
                                        body.substringAfter("[[[\"")
                                            .substringBefore("\"")

                                    runOnUiThread {

                                        resultText.value =
                                            translated
                                    }

                                } catch (
                                    e: Exception
                                ) {

                                    runOnUiThread {

                                        resultText.value =
                                            "Translation parse error"
                                    }
                                }
                            }
                        }
                    })

            } catch (e: Exception) {

                runOnUiThread {

                    resultText.value =
                        "Translation error: ${e.message}"
                }
            }

        }.start()
    }

    override fun onInit(status: Int) {

        if (status == TextToSpeech.SUCCESS) {

            textToSpeech.language = Locale.US
        }
    }

    override fun onDestroy() {

        super.onDestroy()

        textToSpeech.stop()
        textToSpeech.shutdown()
    }
}
